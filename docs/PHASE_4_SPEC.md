# Phase 4 Specification — Flutter Shell

**Your job, Claude Code.** Build a Flutter template that:
- Wraps any website in a native WebView
- Fetches config from AppForge backend at startup
- Works offline after first launch (cache-first)
- Receives push notifications via FCM (shared Firebase project)
- Provides JS bridge for website → native calls (haptic, biometric, share, UPI, analytics)
- Blocks rooted devices (optional, per-app)
- Shows force-update screens (optional, per-app)

## Critical service: RemoteConfigService

**File:** `lib/services/remote_config_service.dart`

Strategy: offline-first.

```dart
class RemoteConfigService {
  static RemoteConfig _current = RemoteConfig.fallback(...);
  static RemoteConfig get current => _current;

  /// Returns true if fresh config was fetched, false if using cache/fallback.
  static Future<bool> initialize({
    String? fcmToken, String? platform, String? deviceModel,
    String? osVersion, String? appVersion,
  }) async {
    // 1. Load cache immediately (fast)
    final cached = await CacheService.loadConfig();
    if (cached != null) _current = cached;

    // 2. Try to fetch fresh
    try {
      final api = ConfigApi(ApiClient(baseUrl: AppConfig.appforgeApiBaseUrl));
      final fresh = await api.fetch(AppConfig.appId,
        fcmToken: fcmToken, platform: platform,
        deviceModel: deviceModel, osVersion: osVersion,
        appVersion: appVersion);
      _current = fresh;
      await CacheService.saveConfig(fresh);
      return true;
    } on NetworkException {
      return false;  // cache/fallback in use
    } on ApiException catch (e) {
      Log.w('[Config] API error ${e.statusCode}');
      return false;
    }
  }

  // Typed accessors
  static String  get appUrl             => _current.appUrl.isNotEmpty ? _current.appUrl : AppConfig.fallbackUrl;
  static String  get themeColor         => _current.theme.primary;
  static String  get accentColor        => _current.theme.accent;
  static String  get splashColor        => _current.splash.color;
  static String  get splashText         => _current.splash.text.isNotEmpty ? _current.splash.text : AppConfig.appName;
  static int     get splashDuration     => _current.splash.durationMs;
  static bool    get splashEnabled      => _current.splash.enabled;
  static String  get splashLogoUrl      => _current.splash.logoUrl;
  static bool    get whatsappEnabled    => _current.features.whatsappShare;
  static bool    get admobEnabled       => _current.features.admob;
  static bool    get biometricEnabled   => _current.features.biometricAuth;
  static bool    get darkModeEnabled    => _current.features.darkMode;
  static bool    get screenshotBlock    => _current.features.screenshotBlock;
  static bool    get rootBlock          => _current.features.rootBlock;
  static bool    get sessionPersistence => _current.features.sessionPersistence;
  static bool    get networkDetection   => _current.features.networkDetection;
  static int     get forceUpdateVersion => _current.forceUpdate.minVersionCode;
  static int     get softUpdateVersion  => _current.softUpdate.minVersionCode;
  static String  get forceUpdateMessage => _current.forceUpdate.message;
  static String  get softUpdateMessage  => _current.softUpdate.message;
  static String  get updateChangelog    => _current.forceUpdate.changelog;
  static Map<String, dynamic> get custom => _current.custom;
}
```

## Critical service: NotificationService

**File:** `lib/services/notification_service.dart`

Key points:
- Subscribes to per-app topic: `app_<uuid-with-underscores>`
- Uses `flutter_local_notifications` for foreground display (FCM doesn't show when app is open)
- Exposes `deepLinkUrl` ValueNotifier — WebViewScreen listens and navigates
- Token + platform sent as query params on next config fetch

```dart
class NotificationService {
  static final _messaging = FirebaseMessaging.instance;
  static final _local = FlutterLocalNotificationsPlugin();
  static final navigatorKey = GlobalKey<NavigatorState>();
  static final deepLinkUrl = ValueNotifier<String?>(null);
  static String? fcmToken;

  static Future<void> initialize() async {
    FirebaseMessaging.onBackgroundMessage(_backgroundHandler);

    // Local notifications for foreground display
    await _local.initialize(
      const InitializationSettings(
        android: AndroidInitializationSettings('@mipmap/ic_launcher'),
        iOS: DarwinInitializationSettings(
          requestAlertPermission: true,
          requestBadgePermission: true,
          requestSoundPermission: true,
        ),
      ),
      onDidReceiveNotificationResponse: (r) {
        if (r.payload != null && r.payload!.isNotEmpty) {
          deepLinkUrl.value = r.payload;
        }
      },
    );

    await _messaging.requestPermission(alert: true, badge: true, sound: true);

    // Per-app topic
    final topic = 'app_${AppConfig.appId.replaceAll("-", "_")}';
    await _messaging.subscribeToTopic(topic);
    await _messaging.subscribeToTopic('all_apps');
    await _messaging.subscribeToTopic(AppConfig.notifChannelTransactional);
    await _messaging.subscribeToTopic(AppConfig.notifChannelAlerts);
    await _messaging.subscribeToTopic(AppConfig.notifChannelPromotional);

    fcmToken = await _messaging.getToken();
    _messaging.onTokenRefresh.listen((t) => fcmToken = t);

    FirebaseMessaging.onMessage.listen((m) {
      _showLocalNotification(m);
      _processDeepLink(m);
    });
    FirebaseMessaging.onMessageOpenedApp.listen(_processDeepLink);

    final initial = await _messaging.getInitialMessage();
    if (initial != null) {
      Future.delayed(const Duration(seconds: 1), () => _processDeepLink(initial));
    }
  }

  // ... _showLocalNotification, _processDeepLink, _backgroundHandler
}
```

## Critical screen: WebViewScreen

**File:** `lib/screens/webview_screen.dart`

The big one. Features to implement:
- Pinch-to-zoom disabled (viewport meta injection)
- Pull-to-refresh (RefreshIndicator)
- Back button: history → exit
- External links → browser
- tel:/mailto:/sms: native
- UPI deep links (gpay, phonepe, paytm, upi schemes)
- Push deep link handling (listens to NotificationService.deepLinkUrl)
- Dark mode CSS class injection on <html>
- JS bridge (window.flutter.haptic/biometric/share/openUPI/track)
- Soft update banner (if buildNumber < soft_update_version)
- Session persistence (last URL, scroll position)
- App rating prompt after N sessions
- WhatsApp FAB (if whatsapp_share enabled)
- Loading indicator overlay while page loads

JS bridge structure (inject in `onPageFinished`):
```js
window.flutter = {
  haptic:    function(t)   { FlutterBridge.postMessage(JSON.stringify({action:'haptic',type:t})); },
  biometric: function(r)   { FlutterBridge.postMessage(JSON.stringify({action:'biometric',reason:r})); },
  share:     function(t,u) { FlutterBridge.postMessage(JSON.stringify({action:'share',text:t,url:u})); },
  openUPI:   function(p)   { FlutterBridge.postMessage(JSON.stringify({action:'upi',params:p})); },
  track:     function(e,p) { FlutterBridge.postMessage(JSON.stringify({action:'track',event:e,props:p||{}})); }
};
```

Handler in `_onJsBridgeMessage`:
```dart
switch (action) {
  case 'haptic':    HapticService.light() / medium / heavy based on type
  case 'biometric': if (config.biometricEnabled) { await BiometricService.authenticate(reason); call JS callback with result }
  case 'share':     launchUrl('https://wa.me/?text=...')
  case 'upi':       try each of gpay/phonepe/paytm/upi schemes, launch first installed
  case 'track':     AnalyticsService.log(event)
}
```

### AppDelegate.swift

```swift
import UIKit
import Flutter
import FirebaseCore
import FirebaseMessaging
import UserNotifications

@main
@objc class AppDelegate: FlutterAppDelegate {
    override func application(_ application: UIApplication,
        didFinishLaunchingWithOptions launchOptions: [UIApplication.LaunchOptionsKey: Any]?) -> Bool {
        FirebaseApp.configure()
        UNUserNotificationCenter.current().delegate = self
        Messaging.messaging().delegate = self
        application.registerForRemoteNotifications()
        GeneratedPluginRegistrant.register(with: self)
        return super.application(application, didFinishLaunchingWithOptions: launchOptions)
    }

    // Screenshot prevention — hide window when app backgrounds
    override func applicationWillResignActive(_ application: UIApplication) {
        window?.isHidden = true
    }
    override func applicationDidBecomeActive(_ application: UIApplication) {
        window?.isHidden = false
    }

    override func application(_ application: UIApplication,
        didRegisterForRemoteNotificationsWithDeviceToken deviceToken: Data) {
        Messaging.messaging().apnsToken = deviceToken
    }
}

extension AppDelegate: MessagingDelegate {
    func messaging(_ messaging: Messaging, didReceiveRegistrationToken fcmToken: String?) {
        print("[FCM] Token: \(fcmToken ?? "nil")")
    }
}
```

## Testing approach after build

1. Run `flutter analyze` — must exit 0 errors
2. Run `flutter test` — must pass (even if it's just the default smoke test)
3. Do NOT build APK yet — wait until backend is deployed
4. Then: fill `firebase_options.dart` via `flutterfire configure`
5. Then: `flutter build apk --release` (with proper signing)
